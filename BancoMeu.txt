CREATE TABLE TIPOINGREDIENTE
	(
		IDTIPOINGREDIENTE INT IDENTITY (1,1) NOT NULL,
		DESCRICAO VARCHAR (500) NOT NULL,
		CONSTRAINT PKTIPOINGREDIENTE PRIMARY KEY (IDTIPOINGREDIENTE)
	)

CREATE TABLE MEDIDA
	(
		IDMEDIDA INT IDENTITY (1,1) NOT NULL,
		NOME VARCHAR(50) NOT NULL,
		DESCRICAO VARCHAR (500) NOT NULL,
		CONSTRAINT PKMEDIDA PRIMARY KEY (IDMEDIDA)
	)

CREATE TABLE BEBIDA
	(
		IDBEBIDA INT IDENTITY (1,1) NOT NULL,
		NOME VARCHAR (50) NOT NULL,
		MODOPREPARO VARCHAR(500) NOT NULL,
		TEMPOPREPARO TIME NOT NULL,
		QUANTIDADE INT NOT NULL,
		IDMEDIDA INT NOT NULL,
		TEMPERATURA INT NOT NULL,
		CONSTRAINT PKBEBIDA PRIMARY KEY (IDBEBIDA),
		CONSTRAINT FKIDMEDIDA FOREIGN KEY (IDMEDIDA) REFERENCES MEDIDA(IDMEDIDA)
	)

CREATE TABLE VENDA
	(
		IDBEBIDA INT NOT NULL,
		DATA DATE NOT NULL,
		QUANTIDADE INT NOT NULL,
		CONSTRAINT PKVENDA PRIMARY KEY (IDBEBIDA),
		CONSTRAINT FKVENDA FOREIGN KEY (IDBEBIDA) REFERENCES BEBIDA(IDBEBIDA)

	)

CREATE TABLE INGREDIENTE
	(
		IDINGREDIENTE INT IDENTITY (1,1) NOT NULL,
		NOME VARCHAR (50) NOT NULL,
		IDTIPOINGREDIENTE INT NOT NULL,
		CONSTRAINT PKINGREDIENTE PRIMARY KEY (IDINGREDIENTE),
		CONSTRAINT FKIDTIPOINGREDIENTE FOREIGN KEY (IDTIPOINGREDIENTE) REFERENCES TIPOINGREDIENTE (IDTIPOINGREDIENTE)
	)

CREATE TABLE BEBIDAINGREDIENTE
	(
		IDBEBIDA INT NOT NULL,
		IDINGREDIENTE INT NOT NULL,
		IDMEDIDA INT NOT NULL,
		QUANTIDADE INT NOT NULL,
		CONSTRAINT PKBEBIDAINGREDIENTE PRIMARY KEY (IDBEBIDA,IDINGREDIENTE),
		CONSTRAINT FKBEBIDAINGREDIENTE FOREIGN KEY (IDMEDIDA) REFERENCES MEDIDA (IDMEDIDA)
	)
